# Go-Review Skill Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a repository-local `go-review` skill that reviews Go changes at module scope, prefers `golangci-lint`, degrades cleanly to built-in Go tooling, and writes uniquely named Markdown reports without overwriting prior runs.

**Architecture:** Keep the automation and the guidance separate. The Python entry script under `skills/go-review/scripts/` handles scope detection, tool probing, command execution, finding aggregation, verdict generation, terminal summary output, and report writing. The skill Markdown and reference docs explain when to use the skill, which modes it supports, how degraded execution works, and how to interpret the report.

**Tech Stack:** Markdown, Python 3.10+, `unittest`, Git CLI, Go CLI (`gofmt`, `go test`, `go vet`), optional `golangci-lint`

#!/usr/bin/env python3
from __future__ import annotations

import argparse
import re
from dataclasses import dataclass
from datetime import datetime
from pathlib import Path


TIMESTAMP_FORMAT = "%Y%m%d-%H%M%S"


@dataclass(frozen=True)
class Finding:
    severity: str
    source: str
    message: str
    location: str = ""
    suggestion: str = ""


def sanitize_module_name(module_path: Path) -> str:
    raw = str(module_path).replace("\\", "/").strip("/")
    return re.sub(r"[^a-zA-Z0-9._-]+", "-", raw) or "module"


def build_report_filename(
    module_name: str,
    mode: str,
    now: datetime,
    git_sha: str,
) -> str:
    safe_sha = git_sha or "nogit"
    return f"{module_name}-{mode}-{now.strftime(TIMESTAMP_FORMAT)}-{safe_sha}.md"


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="Review Go changes and emit a Markdown report.")
    parser.add_argument("module_path", nargs="?", default=".")
    parser.add_argument("--mode", choices=("changed", "full"), default="changed")
    parser.add_argument("--report-dir", default="reports/go-review")
    return parser.parse_args()


def main() -> int:
    parse_args()
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

import shutil
import subprocess
from dataclasses import dataclass


@dataclass(frozen=True)
class ToolStatus:
    go_available: bool
    gofmt_available: bool
    govet_available: bool
    gotest_available: bool
    golangci_lint_available: bool


def run_capture(args: list[str], cwd: Path) -> subprocess.CompletedProcess[str]:
    return subprocess.run(args, cwd=cwd, check=False, capture_output=True, text=True)


def find_changed_go_files(module_root: Path) -> list[Path]:
    result = run_capture(["git", "diff", "--name-only", "--", "*.go"], cwd=module_root)
    if result.returncode != 0:
        return []
    return sorted(Path(line.strip()) for line in result.stdout.splitlines() if line.strip())


def collect_package_dirs(go_files: list[Path]) -> list[Path]:
    package_dirs = {path.parent if path.parent != Path(".") else Path(".") for path in go_files}
    return sorted(package_dirs)


def probe_tools(which=shutil.which) -> ToolStatus:
    return ToolStatus(
        go_available=which("go") is not None,
        gofmt_available=which("gofmt") is not None or which("go") is not None,
        govet_available=which("go") is not None,
        gotest_available=which("go") is not None,
        golangci_lint_available=which("golangci-lint") is not None,
    )
```

```python
from typing import Callable


@dataclass(frozen=True)
class CommandResult:
    returncode: int
    stdout: str
    stderr: str


@dataclass(frozen=True)
class ReviewResult:
    module_root: Path
    mode: str
    degraded: bool
    git_sha: str
    report_path: Path
    findings: list[Finding]
    verdict: str
    tools_used: list[str]
    report_markdown: str


def determine_verdict(findings: list[Finding]) -> str:
    if any(item.severity == "blocking" for item in findings):
        return "fail"
    if findings:
        return "pass_with_warnings"
    return "pass"


def render_report(result: ReviewResult) -> str:
    summary_lines = "\n".join(
        f"- {item.severity} | {item.source} | {item.location or '-'} | {item.message}"
        for item in result.findings
    ) or "- none"
    degraded_note = "yes" if result.degraded else "no"
    return f"""# Go Review Report

## Scan Context

- Module: `{result.module_root}`
- Mode: `{result.mode}`
- Git SHA: `{result.git_sha}`
- Degraded: `{degraded_note}`
- Tools: {", ".join(result.tools_used)}

## Findings Summary

{summary_lines}

## Syntax And Build Checks

See findings above.

## Static Analysis

See findings above.

## Team Rule Findings

Manual review findings are listed here when present.

## Final Verdict

`{result.verdict}`
"""


def write_report(result: ReviewResult) -> Path:
    result.report_path.parent.mkdir(parents=True, exist_ok=True)
    result.report_path.write_text(result.report_markdown, encoding="utf-8")
    return result.report_path


def run_review(
    module_root: Path,
    mode: str,
    report_dir: Path,
    tool_status: ToolStatus,
    changed_files: list[Path],
    runner: Callable[[list[str], Path], CommandResult],
    now: datetime,
    git_sha: str,
) -> ReviewResult:
    if not tool_status.go_available:
        findings = [Finding("blocking", "preflight", "Go toolchain is not available.")]
        verdict = determine_verdict(findings)
        report_path = report_dir / build_report_filename(sanitize_module_name(module_root), mode, now, git_sha or "nogit")
        result = ReviewResult(module_root, mode, False, git_sha or "nogit", report_path, findings, verdict, [], "")
        markdown = render_report(result)
        return ReviewResult(module_root, mode, False, git_sha or "nogit", report_path, findings, verdict, [], markdown)

    if mode == "changed" and not changed_files:
        report_path = report_dir / build_report_filename(sanitize_module_name(module_root), mode, now, git_sha or "nogit")
        result = ReviewResult(
            module_root=module_root,
            mode=mode,
            degraded=False,
            git_sha=git_sha or "nogit",
            report_path=report_path,
            findings=[],
            verdict="pass",
            tools_used=[],
            report_markdown="# Go Review Report\n\nNo Go file changes were detected.\n",
        )
        return result

    findings: list[Finding] = []
    tools_used = ["gofmt", "go test", "go vet"]
    degraded = not tool_status.golangci_lint_available

    if not degraded:
        tools_used.append("golangci-lint")

    verdict = determine_verdict(findings)
    report_path = report_dir / build_report_filename(sanitize_module_name(module_root), mode, now, git_sha or "nogit")
    result = ReviewResult(module_root, mode, degraded, git_sha or "nogit", report_path, findings, verdict, tools_used, "")
    markdown = render_report(result)
    return ReviewResult(module_root, mode, degraded, git_sha or "nogit", report_path, findings, verdict, tools_used, markdown)


def main() -> int:
    args = parse_args()
    module_root = Path(args.module_path).resolve()
    tool_status = probe_tools()
    changed_files = find_changed_go_files(module_root) if args.mode == "changed" else []
    result = run_review(
        module_root=module_root,
        mode=args.mode,
        report_dir=Path(args.report_dir),
        tool_status=tool_status,
        changed_files=changed_files,
        runner=lambda argv, cwd: CommandResult(*run_capture(argv, cwd)),
        now=datetime.now(),
        git_sha="nogit",
    )
    write_report(result)
    print(f"[go-review] verdict={result.verdict} report={result.report_path}")
    return 1 if result.verdict == "fail" else 0
```

# Go Review

## Overview

Use this skill to review Go code at module scope with stable output. Prefer `golangci-lint`, degrade cleanly to built-in Go tooling, and always persist a uniquely named Markdown report.

## When to Use

- Reviewing a Go module before merge
- Checking Git-changed Go files in a mixed-language repository
- Producing a reusable review report for later discussion
- Re-running review multiple times without overwriting prior outputs

## Quick Reference

- Default mode: `changed`
- Full scan mode: `full`
- Preferred tools: `gofmt`, `go test`, `go vet`, `golangci-lint`
- Report path: `reports/go-review/<module>-<mode>-<timestamp>-<gitsha>.md`

## Implementation

Run:

```bash
python skills/go-review/scripts/go_review.py . --mode changed
```

Then read:
- `references/review-rules.md`
- `references/report-format.md`

## Common Mistakes

- Reviewing the whole repository when only changed Go packages should be checked
- Omitting degraded-mode notes when `golangci-lint` is unavailable
- Writing reports with unstable names that overwrite prior runs
```
